    <script>

        document.addEventListener('DOMContentLoaded', () => {

            // --- GAME STATE & CONSTANTS ---

            const STARTING_CASH = 120;

            const COST_PER_UNIT = 8;

            const MAX_WEEKS = 3;

            const PRODUCTS = [

                { name: 'Cool T-Shirt', image: 'https://placehold.co/150x150/e0e7ff/4f46e5?text=T-Shirt' },

                { name: 'Slick Cap', image: 'https://placehold.co/150x150/d1fae5/059669?text=Cap' },

                { name: 'Posh Hoodie', image: 'https://placehold.co/150x150/fee2e2/dc2626?text=Hoodie' }

            ];


            let currentWeek = 1;

            let cash = STARTING_CASH;

            let weeklyHistory = [];

            

            let selectedPrice = null;

            let selectedMarketing = null;


            // --- UI ELEMENTS ---

            const weekDisplay = document.getElementById('weekDisplay');

            const cashDisplay = document.getElementById('cashDisplay');

            const inventoryInput = document.getElementById('inventoryInput');

            const inventoryCostDisplay = document.getElementById('inventoryCostDisplay');

            const endWeekBtn = document.getElementById('endWeekBtn');

            const priceOptions = document.getElementById('priceOptions');

            const marketingOptions = document.getElementById('marketingOptions');

            const productName = document.getElementById('productName');

            const productImage = document.getElementById('productImage');


            // Modals

            const weeklyOutcomeModal = document.getElementById('weeklyOutcomeModal');

            const finalResultsModal = document.getElementById('finalResultsModal');

            const alertModal = document.getElementById('alertModal');

            const alertOkBtn = document.getElementById('alertOkBtn');

            const nextWeekBtn = document.getElementById('nextWeekBtn');

            const restartGameBtn = document.getElementById('restartGameBtn');

            


            // --- EVENT LISTENERS ---

            

            // Handle decision selections

            priceOptions.addEventListener('click', (e) => handleOptionSelect(e, 'price'));

            marketingOptions.addEventListener('click', (e) => handleOptionSelect(e, 'marketing'));


            inventoryInput.addEventListener('input', updateInventoryCost);

            endWeekBtn.addEventListener('click', runWeek);

            nextWeekBtn.addEventListener('click', setupNextWeek);

            restartGameBtn.addEventListener('click', () => location.reload());

            alertOkBtn.addEventListener('click', () => showAlert(null, null, false));



            // --- GAME LOGIC ---


            function initGame() {

                currentWeek = 1;

                cash = STARTING_CASH;

                weeklyHistory = [];

                selectedPrice = null;

                selectedMarketing = null;

                inventoryInput.value = '';

                updateUI();

                clearSelections();

            }


            function handleOptionSelect(event, type) {

                const button = event.target.closest('button');

                if (!button) return;


                const optionsContainer = type === 'price' ? priceOptions : marketingOptions;

                

                // Remove 'selected' class from all buttons in the group

                optionsContainer.querySelectorAll('button').forEach(btn => btn.classList.remove('selected'));

                

                // Add 'selected' class to the clicked button

                button.classList.add('selected');


                if (type === 'price') {

                    selectedPrice = parseInt(button.dataset.price);

                } else if (type === 'marketing') {

                    selectedMarketing = parseInt(button.dataset.marketing);

                }

            }

            

            function updateInventoryCost() {

                const quantity = parseInt(inventoryInput.value) || 0;

                const cost = quantity * COST_PER_UNIT;

                inventoryCostDisplay.textContent = `$${cost}`;

            }


            function runWeek() {

                // 1. Validate Inputs

                const inventoryToBuy = parseInt(inventoryInput.value);

                if (selectedPrice === null) {

                    showAlert('Hold Up, CEO!', 'You must select a selling price.');

                    return;

                }

                if (selectedMarketing === null) {

                    showAlert('Hold Up, CEO!', 'You must choose a marketing spend.');

                    return;

                }

                if (isNaN(inventoryToBuy) || inventoryToBuy < 0) {

                     showAlert('Hold Up, CEO!', 'Please enter a valid quantity for inventory.');

                    return;

                }


                const inventoryCost = inventoryToBuy * COST_PER_UNIT;

                const totalSpend = inventoryCost + selectedMarketing;


                if (totalSpend > cash) {

                    showAlert('Not Enough Cash!', `Your planned spend of $${totalSpend} ($${inventoryCost} for inventory + $${selectedMarketing} for marketing) exceeds your available cash of $${cash}.`);

                    return;

                }


                // 2. Calculate Demand

                const baseDemand = { 5: 50, 10: 25, 15: 12 }[selectedPrice];

                let marketingMultiplier = 1.0;

                const rand = Math.random();


                if (selectedMarketing === 50) {

                    if (rand < 0.7) { // 70% chance

                        marketingMultiplier = 1.0 + Math.random() * 0.3; // 1.0 to 1.3

                    } else { // 30% chance

                        marketingMultiplier = 1.5 + Math.random() * 0.5; // 1.5 to 2.0

                    }

                } else if (selectedMarketing === 100) {

                    if (rand < 0.6) { // 60% chance

                        marketingMultiplier = 1.2 + Math.random() * 0.6; // 1.2 to 1.8

                    } else { // 40% chance

                        marketingMultiplier = 2.0 + Math.random() * 1.0; // 2.0 to 3.0

                    }

                }

                

                const adjustedDemand = Math.round(baseDemand * marketingMultiplier);

                

                // 3. Calculate Financials

                const unitsSold = Math.min(inventoryToBuy, adjustedDemand);

                const revenue = unitsSold * selectedPrice;

                const totalCost = inventoryCost + selectedMarketing;

                const profit = revenue - totalCost;


                // 4. Update Game State

                cash += profit;

                

                const weekResult = {

                    week: currentWeek,

                    product: PRODUCTS[currentWeek - 1].name,

                    price: selectedPrice,

                    marketing: selectedMarketing,

                    unitsBought: inventoryToBuy,

                    unitsSold,

                    revenue,

                    cost: totalCost,

                    profit,

                    endingCash: cash

                };

                weeklyHistory.push(weekResult);


                // 5. Show Outcome

                showWeeklyOutcome(weekResult);

            }


            function setupNextWeek() {

                currentWeek++;

                weeklyOutcomeModal.classList.remove('active');

                if (currentWeek > MAX_WEEKS) {

                    showFinalResults();

                } else {

                    inventoryInput.value = '';

                    selectedPrice = null;

                    selectedMarketing = null;

                    clearSelections();

                    updateUI();

                }

            function awardBadges(totalProfit) {

                const badgesContainer = document.getElementById('badgesContainer');

                badgesContainer.innerHTML = '';

                

                let awarded = false;

                

                if (totalProfit > 0) {

                    badgesContainer.innerHTML += `<div class="badge bg-green-100 text-green-800">💸 Profit Pro</div>`;

                    awarded = true;

                }

                

                const totalUnitsSold = weeklyHistory.reduce((sum, r) => sum + r.unitsSold, 0);

                const totalUnitsBought = weeklyHistory.reduce((sum, r) => sum + r.unitsBought, 0);

                if (totalUnitsBought > 0 && (totalUnitsSold / totalUnitsBought) > 0.85) {

                    badgesContainer.innerHTML += `<div class="badge bg-blue-100 text-blue-800">📊 Efficiency Expert</div>`;

                    awarded = true;

                }

                

                const highMarketingPlays = weeklyHistory.filter(r => r.marketing >= 50 && r.profit > 0).length;

                if (highMarketingPlays >= 2) {

                     badgesContainer.innerHTML += `<div class="badge bg-purple-100 text-purple-800">🧠 Strategic Thinker</div>`;

                     awarded = true;

                }

                

                if (!awarded) {

                    badgesContainer.innerHTML = `<p class="text-gray-500">No badges this time. Keep trying!</p>`;

                }

            }
